CREATE TABLE QUESTION_ANSWER
(
  UNIQUE_ID       VARCHAR(200) NOT NULL COMMENT '대표값',
  QUESTION_ANSWER VARCHAR(200) NOT NULL COMMENT '문항답변',
  PRIMARY KEY (UNIQUE_ID)
) COMMENT '문항답변';

CREATE TABLE RESULT
(
  UNIQUE_ID_SURVEY_LIST     VARCHAR(200) NOT NULL COMMENT '설문목록대표값',
  UNIQUE_ID_SURVEYOR        VARCHAR(200) NOT NULL COMMENT '설문자대표값',
  UNIQUE_ID_QUESTION_ANSWER VARCHAR(200) NOT NULL COMMENT '문항답변대표값'
) COMMENT '결과값';

CREATE TABLE SURVEY_LIST
(
  UNIQUE_ID     VARCHAR(200) NOT NULL COMMENT '대표값',
  QUESTION_LIST VARCHAR(200) NOT NULL COMMENT '문항목록',
  PRIMARY KEY (UNIQUE_ID)
) COMMENT '설문목록';

CREATE TABLE SURVEY
(
  UNIQUE_ID_SURVEY_LIST     VARCHAR(200) NOT NULL COMMENT '설문목록대표값',
  UNIQUE_ID_QUESTION_ANSWER VARCHAR(200) NOT NULL COMMENT '문항답변대표값'
) COMMENT '설문';


CREATE TABLE SURVEYOR

  -- UNIQUE_ID  VARCHAR(200) NOT NULL COMMENT '대표값',    -- 대표값 자동증가가 INT에서만 되서 
  NAME       VARCHAR(200) NOT NULL COMMENT '이름',
  BIRTH_DATE VARCHAR(200) NOT NULL COMMENT '생년월일',
